Références Instruction::_cop, Machine::_data, Instruction::_immediate, Instruction::_regcond, Machine::_registers, Instruction::_value, ADD, address(), error_if_segdata(), Instruction::instr_generic, Instruction::instr_immediate, et set_cc().

Référencé par decode_execute().

{
    unsigned r = instr.instr_generic._regcond;

    if(instr.instr_generic._immediate)
        if(instr.instr_generic._cop == ADD)
            pmach->_registers[r] += instr.instr_immediate._value;

        else
            pmach->_registers[r] -= instr.instr_immediate._value;

    else
    {
        unsigned addr = address(pmach, instr);
        error_if_segdata(pmach, addr);

        if(instr.instr_generic._cop == ADD)
            pmach->_registers[r] += pmach->_data[addr];

        else
            pmach->_registers[r] -= pmach->_data[addr];
    }

    set_cc(pmach, pmach->_registers[r]);
    return true;
}

bool should_jump ( Machine *  pmach,
Instruction  instr 
)

Retourne vrai, si l'on doit sauter false sinon.

Paramètres:
pmachla machine/programme en cours d'exécution
instrl'instruction à exécuter
Renvoie:
true si on doit sauter, false sinon, ne retourne pas si erreur

Définition à la ligne 182 du fichier exec.c.

Références Machine::_cc, Machine::_pc, Instruction::_regcond, CC_N, CC_P, CC_U, CC_Z, EQ, ERR_CONDITION, error(), GE, GT, Instruction::instr_generic, LE, LT, NC, et NE.

Référencé par branch_func(), et call_func().

{
    if((instr.instr_generic._regcond != NC && pmach->_cc == CC_U) ||
            instr.instr_generic._regcond > LAST_CONDITION)
        error(ERR_CONDITION, pmach->_pc - 1);

    switch(instr.instr_generic._regcond)
    {
        case NC:
            return true;

        case EQ:
            return pmach->_cc == CC_Z;

        case NE:
            return pmach->_cc != CC_Z;

        case GT:
            return pmach->_cc == CC_P;

        case GE:
            return pmach->_cc == CC_P || pmach->_cc == CC_Z;

        case LT:
            return pmach->_cc == CC_N;

        case LE:
            return pmach->_cc == CC_N || pmach->_cc == CC_Z;

        default:
            error(ERR_CONDITION, pmach->_pc - 1);
    }
}
